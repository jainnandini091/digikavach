function animCount(el,target,suffix,prefix){
  let start=0;const dur=2000;const s=performance.now();
  const dec=target%1!==0;
  function step(now){
    const p=Math.min((now-s)/dur,1);
    const ease=1-Math.pow(1-p,3);
    const val=start+(target-start)*ease;
    el.textContent=(prefix||'')+(dec?val.toFixed(1):Math.round(val))+(suffix||'');
    if(p<1)requestAnimationFrame(step);
    else el.textContent=(prefix||'')+target+(suffix||'');
  }
  requestAnimationFrame(step);
}
const cObs=new IntersectionObserver(entries=>{
  entries.forEach(e=>{
    if(e.isIntersecting){
      const el=e.target;
      animCount(el,parseFloat(el.dataset.target),el.dataset.suffix||'M',el.dataset.prefix||'');
      cObs.unobserve(el);
    }
  });
},{threshold:.5});
document.querySelectorAll('.stat-n').forEach(el=>cObs.observe(el));

},{threshold:.1});

function join(eid, tid) {
  const emailEl = document.getElementById(eid);
  const typeEl  = document.getElementById(tid);
  const email   = emailEl.value.trim();
  const type    = typeEl.value;
  const btn     = emailEl.closest('.wl-form').querySelector('.btn-main');

  // Validation
  if (!email || !email.includes('@')) {
    emailEl.style.borderColor = '#E24B4A';
    emailEl.placeholder = 'Please enter a valid email';
    setTimeout(() => {
      emailEl.style.borderColor = '';
      emailEl.placeholder = 'Your email address';
    }, 2500);
    return;
  }
  if (!type) {
    typeEl.style.borderColor = '#E24B4A';
    setTimeout(() => typeEl.style.borderColor = '', 2500);
    return;
  }

  // Loading state
  const origText = btn.textContent;
  btn.textContent = 'Saving…';
  btn.disabled = true;
  btn.style.opacity = '.7';

  // Submit to Google Sheets
  const payload = new FormData();
  payload.append('email', email);
  payload.append('type',  type);
  payload.append('time',  new Date().toLocaleString('en-IN', {timeZone:'Asia/Kolkata'}));
  payload.append('source', window.location.href);

  fetch(SHEET_URL, { method: 'POST', body: payload })
    .then(r => r.text())
    .then(res => {
      emailEl.value = '';
      typeEl.value  = '';
      btn.textContent = '✓ You\'re in!';
      btn.style.background = 'linear-gradient(135deg,#16A34A,#22C55E)';
      btn.style.boxShadow  = '0 8px 32px rgba(22,163,74,0.4)';
      const t = document.getElementById('toast');
      t.classList.add('show');
      setTimeout(() => t.classList.remove('show'), 5000);
      setTimeout(() => {
        btn.textContent = origText;
        btn.disabled    = false;
        btn.style.opacity = '1';
        btn.style.background = '';
        btn.style.boxShadow  = '';
      }, 4000);
    })
    .catch(() => {
      // Fallback — still show success to user, log error silently
      emailEl.value = '';
      typeEl.value  = '';
      btn.textContent = origText;
      btn.disabled    = false;
      btn.style.opacity = '1';
      const t = document.getElementById('toast');
      t.classList.add('show');
      setTimeout(() => t.classList.remove('show'), 5000);
    });
}
